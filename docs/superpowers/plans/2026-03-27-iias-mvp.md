# IIAS MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the IIAS (Intelligent Item Authoring System) MVP — a platform that generates, validates, and manages SAT test questions using fine-tuned language models.

**Architecture:** Three-component pipeline: (1) Generation Service fine-tunes Qwen2.5-7B (RW) and Phi-4 (Math) using LoRA, (2) Auto-QA Service validates items via schema checks, readability scoring, and quality rules, (3) Item Bank (PostgreSQL) stores items with lifecycle tracking (draft → pretesting → operational → retired). Components communicate via JSON files and database; no REST API in MVP.

**Tech Stack:** Python 3.11+, PyTorch, Transformers, PEFT/LoRA, PostgreSQL 15, Pydantic v2, textstat, pytest, loguru

```python
from dataclasses import dataclass
from pathlib import Path
from typing import Literal
import yaml
from dotenv import load_dotenv

load_dotenv()

@dataclass
class DatabaseConfig:
    url: str
    pool_size: int = 5
    max_overflow: int = 10

@dataclass
class ModelConfig:
    rw_model_id: str
    math_model_id: str
    fallback_model_id: str = "meta-llama/Llama-3.1-8B-Instruct"

@dataclass
class LoRAConfig:
    r: int = 32
    alpha: int = 64
    dropout: float = 0.05
    target_modules: list = None

    def __post_init__(self):
        if self.target_modules is None:
            self.target_modules = [
                "q_proj", "k_proj", "v_proj", "o_proj",
                "gate_proj", "up_proj", "down_proj"
            ]

@dataclass
class TrainingConfig:
    learning_rate: float = 2e-5
    batch_size: int = 8
    gradient_accumulation_steps: int = 8
    num_epochs: int = 3
    max_seq_length_rw: int = 4096
    max_seq_length_math: int = 2048
    warmup_ratio: float = 0.05
    early_stopping_patience: int = 2

@dataclass
class PathConfig:
    data_dir: Path = Path("data")
    training_dir: Path = Path("data/training")
    generated_dir: Path = Path("data/generated")
    validated_dir: Path = Path("data/validated")
    checkpoint_dir: Path = Path("checkpoints")
    log_dir: Path = Path("outputs/logs")

@dataclass
class Config:
    app_env: Literal["local", "production"]
    database: DatabaseConfig
    models: ModelConfig
    lora: LoRAConfig
    training: TrainingConfig
    paths: PathConfig
    log_level: str = "INFO"

    @classmethod
    def from_yaml(cls, env: Literal["local", "production"] = "local") -> "Config":
        config_file = Path(f"configs/{env}.yaml")
        with open(config_file) as f:
            config_dict = yaml.safe_load(f)

        return cls(
            app_env=env,
            database=DatabaseConfig(**config_dict["database"]),
            models=ModelConfig(**config_dict["models"]),
            lora=LoRAConfig(**config_dict.get("lora", {})),
            training=TrainingConfig(**config_dict.get("training", {})),
            paths=PathConfig(**{k: Path(v) for k, v in config_dict.get("paths", {}).items()}),
            log_level=config_dict.get("log_level", "INFO")
        )

def load_config(env: Literal["local", "production"] = None) -> Config:
    """Load configuration from environment variable or YAML file."""
    import os
    if env is None:
        env = os.getenv("APP_ENV", "local")
    return Config.from_yaml(env)
```

```python
import psycopg2
from psycopg2 import pool
from contextlib import contextmanager
from loguru import logger
from src.config import load_config

class Database:
    """Database connection manager."""

    def __init__(self):
        self.connection_pool = None
        self._config = None

    def initialize(self, config):
        """Initialize connection pool from config."""
        self._config = config
        self.connection_pool = pool.SimpleConnectionPool(
            minconn=1,
            maxconn=config.database.pool_size,
            dsn=config.database.url
        )
        logger.info(f"Database connection pool created: {config.database.url}")

    @contextmanager
    def get_connection(self):
        """Get a connection from the pool."""
        conn = self.connection_pool.getconn()
        try:
            yield conn
            conn.commit()
        except Exception:
            conn.rollback()
            raise
        finally:
            self.connection_pool.putconn(conn)

    def close_all(self):
        """Close all connections."""
        if self.connection_pool:
            self.connection_pool.closeall()
            logger.info("Database connection pool closed")

# Global instance
db = Database()
```

```python
from pydantic import BaseModel, Field, field_validator
from typing import Literal, Optional, List
from enum import Enum
import uuid

class Section(str, Enum):
    READING_WRITING = "reading_writing"
    MATH = "math"

class Difficulty(str, Enum):
    EASY = "easy"
    MEDIUM = "medium"
    HARD = "hard"

class MathFormat(str, Enum):
    PLAIN = "plain"
    LATEX = "latex"

class Choice(BaseModel):
    label: Literal["A", "B", "C", "D"]
    text: str

class ContentJSON(BaseModel):
    passage: Optional[str] = None
    question: str
    math_format: MathFormat = MathFormat.PLAIN
    choices: List[Choice]
    correct_answer: Literal["A", "B", "C", "D"]
    correct_answer_text: str
    rationale: str
    solution_steps: Optional[str] = None

    @field_validator("choices")
    @classmethod
    def validate_choices(cls, v):
        if len(v) != 4:
            raise ValueError("Must have exactly 4 choices")
        expected_labels = ["A", "B", "C", "D"]
        actual_labels = [choice.label for choice in v]
        if actual_labels != expected_labels:
            raise ValueError(f"Choices must be labeled {expected_labels} in order")
        return v

    @field_validator("correct_answer")
    @classmethod
    def validate_correct_answer(cls, v, info):
        choices = {choice.label: choice.text for choice in info.data.get("choices", [])}
        if v not in choices:
            raise ValueError(f"correct_answer {v} not in choices")
        return v

    @field_validator("correct_answer_text")
    @classmethod
    def validate_correct_answer_text(cls, v, info):
        correct_answer = info.data.get("correct_answer")
        choices = {choice.label: choice.text for choice in info.data.get("choices", [])}
        if correct_answer and choices.get(correct_answer) != v:
            raise ValueError("correct_answer_text must match text of correct_answer choice")
        return v

class Item(BaseModel):
    id: str
    section: Section
    domain: str
    difficulty: Difficulty
    content_json: ContentJSON
    model_version: Optional[str] = None

    @field_validator("id")
    @classmethod
    def validate_id(cls, v):
        try:
            uuid.UUID(v)
        except ValueError:
            raise ValueError("id must be a valid UUID")
        return v

class AutoQAResult(BaseModel):
    item_id: str
    validation_timestamp: str
    schema_valid: bool
    auto_qa_passed: bool
    qa_score: float
    checks: dict
    qa_flags: List[str]
```

```python
import textstat
from typing import Optional

class ReadabilityChecker:
    """Check passage readability."""

    def __init__(self, min_grade: float = 9.0, max_grade: float = 12.0):
        self.min_grade = min_grade
        self.max_grade = max_grade

    def check(self, item: dict) -> dict:
        """
        Check readability of passage.

        Returns:
            {"passed": bool, "grade_level": float, "word_count": int, "issues": list}
        """
        passage = item.get("content_json", {}).get("passage")

        if passage is None:
            return {"passed": True, "note": "No passage"}

        try:
            grade_level = textstat.flesch_kincaid_grade(passage)
            word_count = textstat.word_count(passage)
        except:
            return {"passed": True, "note": "Could not calculate readability"}

        issues = []
        if grade_level < self.min_grade:
            issues.append(f"Grade level {grade_level:.1f} below {self.min_grade}")
        elif grade_level > self.max_grade:
            issues.append(f"Grade level {grade_level:.1f} above {self.max_grade}")

        return {
            "passed": len(issues) == 0,
            "grade_level": grade_level,
            "word_count": word_count,
            "issues": issues
        }
```

```python
import re
from typing import List

class QualityRulesChecker:
    """Check basic quality rules."""

    def check(self, item: dict) -> dict:
        """
        Check basic quality rules.

        Returns:
            {"passed": bool, "issues": list}
        """
        issues = []

        # Check duplicate choices
        if not self._check_duplicate_choices(item):
            issues.append("Duplicate choice texts found")

        # Check question length
        if not self._check_question_length(item):
            issues.append("Question too short (< 20 characters)")

        # Check rationale length
        if not self._check_rationale_length(item):
            issues.append("Rationale too short (< 30 words)")

        # Check LaTeX for Math
        latex_check = self._check_math_latex(item)
        if latex_check.get("applicable") and not latex_check.get("passed"):
            issues.append(latex_check.get("error"))

        return {"passed": len(issues) == 0, "issues": issues}

    def _check_duplicate_choices(self, item: dict) -> bool:
        """Check for duplicate choice texts."""
        choices = item.get("content_json", {}).get("choices", [])
        if not choices:
            return False
        texts = [choice.get("text", "") for choice in choices]
        return len(texts) == len(set(texts))

    def _check_question_length(self, item: dict) -> bool:
        """Check if question meets minimum length."""
        question = item.get("content_json", {}).get("question", "")
        return len(question.strip()) >= 20

    def _check_rationale_length(self, item: dict) -> bool:
        """Check if rationale meets minimum word count."""
        rationale = item.get("content_json", {}).get("rationale", "")
        return len(rationale.split()) >= 30

    def _check_math_latex(self, item: dict) -> dict:
        """Check if Math item uses LaTeX appropriately."""
        if item.get("section") != "math":
            return {"applicable": False}

        question = item.get("content_json", {}).get("question", "")
        math_format = item.get("content_json", {}).get("math_format")

        # Detect math expressions
        has_expressions = bool(re.search(r'[a-z]\s*[\^=]', question))

        if has_expressions and math_format != "latex":
            return {
                "applicable": True,
                "passed": False,
                "error": "Math item has expressions but math_format is not 'latex'"
            }

        return {"applicable": True, "passed": True}
```

```python
import uuid
from datetime import datetime
from loguru import logger
from src.auto_qa.validators.schema_validator import SchemaValidator
from src.auto_qa.validators.readability_checker import ReadabilityChecker
from src.auto_qa.validators.quality_rules import QualityRulesChecker

class AutoQAPipeline:
    """Main Auto-QA validation pipeline."""

    def __init__(self):
        self.schema_validator = SchemaValidator()
        self.readability_checker = ReadabilityChecker()
        self.quality_rules_checker = QualityRulesChecker()

    def validate(self, item: dict) -> dict:
        """
        Run full Auto-QA pipeline on item.

        Returns:
            Validation result dict
        """
        result = {
            "item_id": item.get("id"),
            "validation_timestamp": datetime.utcnow().isoformat(),
            "schema_valid": None,
            "auto_qa_passed": None,
            "qa_score": 0.0,
            "checks": {},
            "qa_flags": []
        }

        # Stage 1: Schema validation (hard gate)
        schema_result = self.schema_validator.validate(item)
        result["checks"]["schema_validation"] = schema_result
        result["schema_valid"] = schema_result["passed"]

        if not schema_result["passed"]:
            result["auto_qa_passed"] = False
            result["qa_flags"].extend(schema_result["issues"])
            return result

        # Stage 2: Readability check (warning gate)
        readability_result = self.readability_checker.check(item)
        result["checks"]["readability"] = readability_result

        if not readability_result.get("passed", True):
            result["qa_flags"].extend(readability_result.get("issues", []))

        # Stage 3: Quality rules (hard gate)
        quality_result = self.quality_rules_checker.check(item)
        result["checks"]["quality_rules"] = quality_result

        if not quality_result["passed"]:
            result["auto_qa_passed"] = False
            result["qa_flags"].extend(quality_result["issues"])
            return result

        # All checks passed
        result["auto_qa_passed"] = True
        result["qa_score"] = 1.0

        return result
```

```python
import json
import uuid
from pathlib import Path
from loguru import logger
from datetime import datetime

def convert_itembank(source_path: str, output_dir: str):
    """
    Convert ItemBank format to Meridian training format.

    ItemBank format:
    {
      "id": "uuid",
      "section": "reading_writing|math",
      "domain": "string",
      "difficulty_tier": "easy|medium|hard",
      "content_json": {...}
    }

    Training format (chat template):
    {
      "dataset_version": "rw-sft-v1.0",
      "schema_version": "item-schema-v1",
      "section": "reading_writing|math",
      "domain": "string",
      "difficulty_tier": "easy|medium|hard",
      "messages": [
        {"role": "system", "content": "..."},
        {"role": "user", "content": "Generate..."},
        {"role": "assistant", "content": "{...item...}"}
      ]
    }
    """
    source_file = Path(source_path)
    output_path = Path(output_dir)
    output_path.mkdir(parents=True, exist_ok=True)

    logger.info(f"Converting {source_file}...")

    with open(source_file) as f:
        items = json.load(f)

    rw_items = []
    math_items = []

    for item in items:
        # Map section names
        section = item.get("section", "").replace("reading_writing", "rw").replace("math", "math")
        if section not in ["rw", "math"]:
            continue

        # Map difficulty
        difficulty = item.get("difficulty_tier", "medium")

        # Extract domain and topic
        domain = item.get("domain", "")
        topic = item.get("content_json", {}).get("metadata", {}).get("topic", "")

        # Build messages
        messages = [
            {
                "role": "system",
                "content": "You are an expert SAT item writer. Generate items in strict JSON matching the provided schema. Output JSON only, with no preamble, explanation, or markdown formatting."
            },
            {
                "role": "user",
                "content": f"Generate a single SAT {'Reading & Writing' if section == 'rw' else 'Math'} item.\n\nConstraints:\n- section: {section}\n- domain: {domain}\n- difficulty_tier: {difficulty}" + (f"\n- topic: {topic}" if topic else "")
            },
            {
                "role": "assistant",
                "content": json.dumps(item["content_json"], indent=2)
            }
        ]

        training_example = {
            "dataset_version": f"{section}-sft-v1.0",
            "schema_version": "item-schema-v1",
            "section": section,
            "domain": domain,
            "difficulty_tier": difficulty,
            "messages": messages
        }

        if section == "rw":
            rw_items.append(training_example)
        else:
            math_items.append(training_example)

    # Split into train/val (85/15)
    def split(items):
        split_idx = int(len(items) * 0.85)
        return items[:split_idx], items[split_idx:]

    rw_train, rw_val = split(rw_items)
    math_train, math_val = split(math_items)

    # Write JSONL files
    def write_jsonl(items, path):
        with open(path, "w") as f:
            for item in items:
                f.write(json.dumps(item) + "\n")

    write_jsonl(rw_train, output_path / "rw_train.jsonl")
    write_jsonl(rw_val, output_path / "rw_val.jsonl")
    write_jsonl(math_train, output_path / "math_train.jsonl")
    write_jsonl(math_val, output_path / "math_val.jsonl")

    logger.info(f"✅ Conversion complete:")
    logger.info(f"  RW: {len(rw_train)} train, {len(rw_val)} val")
    logger.info(f"  Math: {len(math_train)} train, {len(math_val)} val")

if __name__ == "__main__":
    import sys
    if len(sys.argv) < 3:
        print("Usage: python convert_itembank.py <source_path> <output_dir>")
        sys.exit(1)

    convert_itembank(sys.argv[1], sys.argv[2])
```

```python
                model_version
            ) VALUES (
                %s, %s, %s, %s,
                %s, %s, %s,
                %s, %s, %s, %s,
                %s,
                %s, %s, %s,

            item_id,

            datetime.utcnow(),
            item['section'],
            item['domain'],
            item['difficulty'],

            qa_result.get('auto_qa_passed', False),
            qa_result.get('qa_score', 0.0),
            qa_result.get('qa_flags', []),
            item.get('model_version', 'unknown')
        ))

        logger.info(f"Item created: {item_id}")
        return item_id

    def get_by_id(self, item_id: str, conn):
        """Get item by ID."""
        cur = conn.cursor()
        cur.execute("SELECT * FROM items WHERE id = %s", (item_id,))
        return cur.fetchone()

    def query(self, conn, **filters):
        """Query items with filters."""
        query = "SELECT * FROM items WHERE 1=1"
        params = []

        if "status" in filters:
            query += " AND status = %s"
            params.append(filters["status"])

        if "section" in filters:
            query += " AND section = %s"
            params.append(filters["section"])

        if "domain" in filters:
            query += " AND domain = %s"
            params.append(filters["domain"])

        query += " ORDER BY created_at DESC"

        if "limit" in filters:
            query += " LIMIT %s"
            params.append(filters["limit"])

        cur = conn.cursor()
        cur.execute(query, params)
        return cur.fetchall()

    def update_status(self, item_id: str, new_status: str, conn, reviewer_id: str = None):
        """Update item status."""
        cur = conn.cursor()

        # Get old status
        cur.execute("SELECT status FROM items WHERE id = %s", (item_id,))
        result = cur.fetchone()
        if not result:
            raise ValueError(f"Item {item_id} not found")
        old_status = result[0]

        # Update status
        cur.execute("""
            UPDATE items
            SET status = %s, updated_at = %s
            WHERE id = %s
        """, (new_status, datetime.utcnow(), item_id))

        # Log status change
        cur.execute("""
            INSERT INTO audit_log (log_id, item_id, timestamp, action_type, actor_type, actor_id, before_status, after_status)
            VALUES (%s, %s, %s, %s, %s, %s, %s, %s)
        """, (
            str(uuid.uuid4()),
            item_id,
            datetime.utcnow(),
            f"status_change_to_{new_status}",
            'user' if reviewer_id else 'system',
            reviewer_id,
            old_status,
            new_status
        ))

        logger.info(f"Item {item_id}: {old_status} → {new_status}")
```

```python
import torch
from torch.utils.data import Dataset
from typing import List, Dict
from pathlib import Path
import json

class SATDataset(Dataset):
    """Dataset for SAT item generation training."""

    def __init__(
        self,
        jsonl_path: str,
        tokenizer,
        max_length: int = 4096,
        section: str = "reading_writing"
    ):
        self.data = []
        self.tokenizer = tokenizer
        self.max_length = max_length
        self.section = section

        # Load JSONL
        with open(jsonl_path) as f:
            for line in f:
                self.data.append(json.loads(line))

    def __len__(self):
        return len(self.data)

    def __getitem__(self, idx: int) -> Dict[str, torch.Tensor]:
        example = self.data[idx]
        messages = example["messages"]

        # Apply chat template
        text = self.tokenizer.apply_chat_template(
            messages,
            tokenize=False,
            add_generation_prompt=False
        )

        # Tokenize
        encodings = self.tokenizer(
            text,
            max_length=self.max_length,
            padding="max_length",
            truncation=True,
            return_tensors="pt"
        )

        # Create labels with loss masking
        # Mask all tokens except assistant turn
        labels = encodings["input_ids"].clone()

        # Find assistant message start in text
        # Simplified: find "assistant" marker and mask everything before it
        assistant_marker = self.tokenizer.encode("assistant", add_special_tokens=False)
        assistant_start = None
        for i in range(len(labels) - len(assistant_marker)):
            if torch.equal(labels[i:i+len(assistant_marker)], assistant_marker.to(labels.device)):
                assistant_start = i + len(assistant_marker)
                break

        if assistant_start is not None:
            labels[:assistant_start] = -100  # Mask non-assistant tokens

        return {
            "input_ids": encodings["input_ids"].squeeze(0),
            "attention_mask": encodings["attention_mask"].squeeze(0),
            "labels": labels.squeeze(0)
        }
```

```python
        target_modules=config.lora.target_modules,
```

```python
import json
from typing import List
from loguru import logger
from src.generation.models import load_model_for_training
from src.config import load_config
from peft import PeftModel

class ItemGenerator:
    """Generate SAT items using fine-tuned model."""

    def __init__(self, section: str, checkpoint_path: str):
        """
        Initialize generator.

        Args:
            section: "reading_writing" or "math"
            checkpoint_path: Path to LoRA adapter checkpoint
        """
        self.section = section
        self.checkpoint_path = checkpoint_path

        logger.info(f"Loading model for {section} from {checkpoint_path}")
        config = load_config()

        # Load base model
        base_model_id = config.models.rw_model_id if section == "reading_writing" else config.models.math_model_id
        from transformers import AutoModelForCausalLM, AutoTokenizer
        from transformers import BitsAndBytesConfig

        tokenizer = AutoTokenizer.from_pretrained(base_model_id)
        base_model = AutoModelForCausalLM.from_pretrained(
            base_model_id,
            device_map="auto",
            quantization_config=BitsAndBytesConfig(load_in_4bit=True)
        )

        # Load LoRA adapter
        self.model = PeftModel.from_pretrained(base_model, checkpoint_path)
        self.tokenizer = tokenizer

        logger.info(f"Model loaded for {section}")

    def generate(self, domain: str, difficulty: str, topic: str = None, count: int = 1) -> List[dict]:
        """
        Generate SAT items.

        Args:
            domain: e.g., "algebra.quadratic_equations"
            difficulty: "easy", "medium", or "hard"
            topic: Optional specific topic
            count: Number of items to generate

        Returns:
            List of generated items (dict format)
        """
        logger.info(f"Generating {count} {self.section} items: {domain}, {difficulty}")

        items = []
        for _ in range(count):
            # Build prompt
            system_msg = "You are an expert SAT item writer. Generate items in strict JSON matching the provided schema. Output JSON only, with no preamble, explanation, or markdown formatting."

            user_msg = f"Generate a single SAT {'Reading & Writing' if self.section == 'reading_writing' else 'Math'} item.\n\nConstraints:\n- section: {self.section}\n- domain: {domain}\n- difficulty_tier: {difficulty}"
            if topic:
                user_msg += f"\n- topic: {topic}"

            messages = [
                {"role": "system", "content": system_msg},
                {"role": "user", "content": user_msg}
            ]

            # Apply chat template
            prompt = self.tokenizer.apply_chat_template(
                messages,
                tokenize=False,
                add_generation_prompt=True
            )

            # Generate
            inputs = self.tokenizer(prompt, return_tensors="pt").to(self.model.device)
            outputs = self.model.generate(
                **inputs,
                max_new_tokens=2048,
                temperature=0.7,
                top_p=0.9,
                do_sample=True,
                pad_token_id=self.tokenizer.eos_token_id
            )

            # Decode
            generated_text = self.tokenizer.decode(outputs[0][inputs["input_ids"].shape[1]:], skip_special_tokens=True)

            # Parse JSON
            try:
                item_json = json.loads(generated_text)
                item_json["section"] = self.section
                item_json["domain"] = domain
                item_json["difficulty"] = difficulty
                item_json["model_version"] = checkpoint_path
                items.append(item_json)
            except json.JSONDecodeError:
                logger.warning(f"Failed to parse generated JSON: {generated_text[:100]}...")

        logger.info(f"✅ Generated {len(items)} items")
        return items
```

```python
from loguru import logger
from src.auto_qa.pipeline import AutoQAPipeline

def validate_items(input_path: str, output_dir: str = None):
    """Run Auto-QA on generated items."""
    pipeline = AutoQAPipeline()

    input_file = Path(input_path)
    output_dir = Path(output_dir) if output_dir else Path("data/validated")
    output_dir.mkdir(parents=True, exist_ok=True)

    passed_items = []
    failed_items = []

    with open(input_file) as f:
        for line_num, line in enumerate(f, 1):
            try:
                item = json.loads(line)

                # Run validation
                result = pipeline.validate(item)

                if result["auto_qa_passed"]:
                    passed_items.append(item)
                else:
                    failed_items.append({"item": item, "result": result})

            except json.JSONDecodeError as e:
                logger.error(f"Line {line_num}: Invalid JSON - {e}")
                failed_items.append({"line_number": line_num, "error": str(e)})

    # Save passed items
    if passed_items:
        output_file = output_dir / "validated_items.jsonl"
        with open(output_file, "w") as f:
            for item in passed_items:
                f.write(json.dumps(item) + "\n")
        logger.info(f"✅ Saved {len(passed_items)} validated items to {output_file}")

    # Save failed items for review
    if failed_items:
        rejected_file = output_dir / "rejected_items.jsonl"
        with open(rejected_file, "w") as f:
            for item in failed_items:
                f.write(json.dumps(item) + "\n")
        logger.warning(f"⚠️  {len(failed_items)} items rejected. See {rejected_file}")

    logger.info(f"Validation complete: {len(passed_items)} passed, {len(failed_items)} failed")

if __name__ == "__main__":
    if len(sys.argv) < 2:
        print("Usage: python validate_items.py <input_jsonl> [output_dir]")
        sys.exit(1)

    validate_items(sys.argv[1], sys.argv[2] if len(sys.argv) > 2 else None)
```
